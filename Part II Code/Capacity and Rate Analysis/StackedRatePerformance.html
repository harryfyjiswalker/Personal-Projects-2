    '/Users/harryfyjis-walker/Desktop/Part II Stuff/Electrochem Data for analaysis/HFW-RATE-NFMMO-552850-REDO-2_1.5-4.3_NEWARE_3-8_MB1.51_COPYDATA (1).xlsx',

]

# Group Excel files based on their filenames
file_groups = [
    excel_files[:3],  # Files starting with '282'
    excel_files[3:6], # Files starting with '552'
    excel_files[6:],  # Files starting with '822'
]

# Define different shades of blue
colors = {
    '282750': 'lawngreen',
    '282800': 'darkseagreen',
    '282850': 'darkgreen',
    '552750': 'skyblue',
    '552800': 'cornflowerblue',
    '552850': 'darkblue',
    '822800': 'mediumvioletred',
    '822850': 'indigo'
}

# Calculate the total height required for the figure based on the number of groups
total_height = 4 * len(file_groups)

# Create a new figure with the calculated total height
fig, axs = plt.subplots(nrows=len(file_groups), ncols=1, figsize=(8, total_height), sharex=True, sharey=True)

# Plot each group of Excel files separately
for i, group in enumerate(file_groups):
    # Initialize lists to store all x and y values
    all_x_values = []
    all_y_values = []
    file_labels = []

    # Iterate over each Excel file in the group and plot the data
    for excel_file in group:
        # Extract six consecutive numbers from the filename
        numbers = re.findall(r'\d{6}', excel_file)
        if numbers:
            label = numbers[0]  # Take the first set of six consecutive numbers as the label
            file_labels.append(label)
        else:
            file_labels.append("Unknown")

        # Load the Excel file into a DataFrame
        df = pd.read_excel(excel_file)
        
        # Initialize lists to store x and y values for the current file
        x_values = []
        y_values = []

        # Iterate over columns and find the final non-zero value for each odd-numbered column
        for col in df.columns:
            if 'Col' in col and int(col.split('Col')[-1]) % 2 != 0:
                # Find the last non-zero and non-NaN value in the column
                non_zero_values = df[col][(df[col] != 0) & (~df[col].isnull())]
                if not non_zero_values.empty:
                    final_value = non_zero_values.iloc[-1]
                    cycle_number = (int(col.split('Col')[-1]) + 1) / 2
                    if cycle_number <= 98:  # Only append values up to cycle number 98
                        x_values.append(final_value)
                        y_values.append(cycle_number)  # Calculate the corresponding y-value
        
        # Append the x and y values for the current file to the lists storing all values
        all_x_values.append(x_values)
        all_y_values.append(y_values)

    # Perform cubic spline interpolation for each file separately
    interpolated_curves = []
    for j in range(len(group)):
        # Perform cubic spline interpolation
        f = interp1d(all_y_values[j], all_x_values[j], kind='cubic')
        x_new = np.linspace(min(all_y_values[j]), 74, 100)  # Adjust range to include only up to cycle number 98
        y_new = f(x_new)
        interpolated_curves.append((x_new, y_new))

    # Plot all the interpolated curves on the corresponding subplot with unique colors and labels
    for k in range(len(interpolated_curves)):
        x_new, y_new = interpolated_curves[k]
        axs[i].plot(x_new, y_new, color=colors[file_labels[k]], label=file_labels[k])
        
    # Add grid to the subplot
    axs[i].grid(True)

    # Add legend to the subplot
    axs[i].legend(loc='lower right')

# Add a single y-axis label for the entire figure
fig.text(0.04, 0.5, 'Specific Discharge Capacity / mAh/g', va='center', rotation='vertical')

# Add labels and title to the shared x-axis
axs[-1].set_xlabel('Cycle Number')

plt.ylim(1, 150)
plt.xlim(1, 100)

plt.subplots_adjust(hspace=0)

  # Adjust subplots to prevent overlap
